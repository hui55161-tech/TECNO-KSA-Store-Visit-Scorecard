// ============================================================ // TECNO KSA · Store Visit Scorecard v3 — Google Apps Script // 适配现有 Visits Sheet 列顺序： // A=id, B=store, C=promoter, D=star, E=auditor, F=date, G=time, // H=total, I=grade, J=A_display, K=B_materials, L=C_skills, // M=D_ops, N=E_service, O=highlights, P=actions, Q=feedback, // R=auditor_name, S=created_at // 新增列从 T 开始追加：store_type, chain, ba, region, city, // total_norm, X_visibility, Y_brand, Z_stock, R_relationship, item_scores // ============================================================ var SHEET_NAME = 'Visits'; function getSheet() { var ss = SpreadsheetApp.getActiveSpreadsheet(); var sheet = ss.getSheetByName(SHEET_NAME); if (!sheet) sheet = ss.insertSheet(SHEET_NAME); return sheet; } // 读取表头，返回 {列名: 列索引(0-based)} 映射 function getHeaderMap(sheet) { var lastCol = sheet.getLastColumn(); if (lastCol < 1) return {}; var headers = sheet.getRange(1, 1, 1, lastCol).getValues()[0]; var map = {}; headers.forEach(function(h, i) { if (h) map[String(h).trim()] = i; }); return map; } // 确保所有需要的列都存在，缺失的追加到末尾 function ensureColumns(sheet) { var required = [ 'id','store','promoter','star','auditor','date','time', 'total','grade','A_display','B_materials','C_skills', 'D_ops','E_service','highlights','actions','feedback', 'auditor_name','created_at', // 扩展列（新增） 'store_type','chain','ba','region','city', 'total_norm','X_visibility','Y_brand','Z_stock','R_relationship','item_scores' ]; var map = getHeaderMap(sheet); var missing = required.filter(function(h){ return map[h] === undefined; }); if (missing.length > 0) { var lastCol = sheet.getLastColumn(); var newHeaders = sheet.getRange(1, lastCol+1, 1, missing.length); newHeaders.setValues([missing]); newHeaders.setFontWeight('bold'); // 重新读取 map = getHeaderMap(sheet); } return map; } // 从 secs[] 或 itemScores{} 提取各维度得分 function extractSecScore(rec, secId) { if (rec.secs && Array.isArray(rec.secs)) { var s = rec.secs.find(function(x){ return x.id === secId; }); if (s && Number(s.score) > 0) return Number(s.score); } if (rec.itemScores && typeof rec.itemScores === 'object') { var total = 0; Object.keys(rec.itemScores).forEach(function(k){ if (k.split('_')[0] === secId) total += Number(rec.itemScores[k]) || 0; }); if (total > 0) return total; } return 0; } // 日期格式化为 "YYYY-MM-DD" 字符串 function formatDate(val) { if (!val && val !== 0) return ''; if (val instanceof Date) { if (val.getFullYear() < 1970) return ''; return val.getFullYear() + '-' + String(val.getMonth()+1).padStart(2,'0') + '-' + String(val.getDate()).padStart(2,'0'); } if (typeof val === 'string') { if (val.startsWith('1899') || val.startsWith('1900')) return ''; if (/^\d{4}-\d{2}-\d{2}/.test(val)) return val.slice(0,10); if (/^\d{4}\/\d{2}\/\d{2}/.test(val)) return val.slice(0,10).replace(/\//g,'-'); } if (typeof val === 'number' && val > 0 && val < 100000) { var d = new Date(Date.UTC(1899,11,30) + val*86400000); if (d.getFullYear() < 1970) return ''; return d.getFullYear() + '-' + String(d.getMonth()+1).padStart(2,'0') + '-' + String(d.getDate()).padStart(2,'0'); } return ''; } function todayStr() { var tz = Session.getScriptTimeZone(); return Utilities.formatDate(new Date(), tz, 'yyyy-MM-dd'); } function jsonOut(data) { return ContentService .createTextOutput(JSON.stringify(data)) .setMimeType(ContentService.MimeType.JSON); } // ── doPost ────────────────────────────────────────────────── function doPost(e) { try { var body = JSON.parse(e.postData.contents); var rec = body.record; if (!rec) return jsonOut({ok:false, error:'No record'}); var sheet = getSheet(); var hMap = ensureColumns(sheet); var totalCols = sheet.getLastColumn(); // 各维度得分 var aScore = extractSecScore(rec,'A'); var bScore = extractSecScore(rec,'B'); var cScore = extractSecScore(rec,'C'); var dScore = extractSecScore(rec,'D'); var eScore = extractSecScore(rec,'E'); var xScore = extractSecScore(rec,'X'); var yScore = extractSecScore(rec,'Y'); var zScore = extractSecScore(rec,'Z'); var rScore = extractSecScore(rec,'R'); var isP = rec.store_type !== 'nonpromoter'; var rawFromSecs = isP ? (aScore+bScore+cScore+dScore+eScore) : (xScore+yScore+zScore+rScore); var totalRaw = rawFromSecs || Number(rec.total_raw) || 0; var totalNorm = isP ? totalRaw : Math.round(totalRaw/55*100); // 日期 var dateStr = ''; if (rec.date && String(rec.date).length >= 10 && !String(rec.date).startsWith('1899')) { dateStr = String(rec.date).slice(0,10); } if (!dateStr) dateStr = todayStr(); // 字段值映射（支持新旧两套列名） var fields = { 'id': String(rec.id || Date.now()), 'store': rec.store || '', 'store_type': rec.store_type || 'promoter', 'chain': rec.chain || '', 'ba': rec.ba || '', 'promoter': rec.promoter || '', 'star': rec.star || '', 'region': rec.region || '', 'city': rec.city || '', 'auditor': rec.auditor || '', 'auditor_name': rec.auditor_name || rec.auditor || '', 'date': dateStr, 'time':(()=>{const t=rec.time||'';if(!t)return'';if(/^\d{2}:\d{2}/.test(t))return t.slice(0,5);const m=t.match(/(\d{2}:\d{2}):\d{2}/);return m?m[1]:t.slice(0,5);})(), 'total': totalRaw, 'total_norm': totalNorm, 'grade': rec.grade || '', 'A_display': aScore, 'B_materials': bScore, 'C_skills': cScore, 'D_ops': dScore, 'E_service': eScore, 'X_visibility': xScore, 'Y_brand': yScore, 'Z_stock': zScore, 'R_relationship': rScore, 'highlights': rec.h1 || '', 'actions': rec.h2 || '', 'feedback': rec.h3 || '', 'item_scores': rec.itemScores ? JSON.stringify(rec.itemScores) : '', 'created_at': new Date().toISOString() }; // 按列索引填充行 var row = new Array(totalCols).fill(''); Object.keys(fields).forEach(function(f){ if (hMap[f] !== undefined) row[hMap[f]] = fields[f]; }); sheet.appendRow(row); // 日期列设为文本格式 if (hMap['date'] !== undefined) { var lastRow = sheet.getLastRow(); sheet.getRange(lastRow, hMap['date']+1).setNumberFormat('@STRING@'); } return jsonOut({ok:true, id:fields['id']}); } catch(err) { return jsonOut({ok:false, error:err.message}); } } // ── doGet ─────────────────────────────────────────────────── function doGet(e) { try { var sheet = getSheet(); var lastRow = sheet.getLastRow(); var lastCol = sheet.getLastColumn(); if (lastRow <= 1 || lastCol < 1) return jsonOut({ok:true, records:[]}); var data = sheet.getRange(1, 1, lastRow, lastCol).getValues(); var hdrs = data[0].map(function(h){ return String(h).trim(); }); var numCols = ['total','total_raw','total_norm', 'A_display','B_materials','C_skills','D_ops','E_service', 'X_visibility','Y_brand','Z_stock','R_relationship']; var records = []; for (var i = 1; i < data.length; i++) { var row = data[i]; var obj = {}; hdrs.forEach(function(h, idx){ var v = row[idx]; if (h === 'date') { obj[h] = formatDate(v); } else if (h === 'created_at') { obj[h] = v instanceof Date ? v.toISOString() : String(v||''); } else if (numCols.indexOf(h) >= 0) { obj[h] = Number(v) || 0; } else { obj[h] = (v === null || v === undefined) ? '' : String(v); } }); // 兼容旧字段名：total → total_raw / total_norm if (!obj.total_raw && obj.total) obj.total_raw = obj.total; if (!obj.total_norm && obj.total) obj.total_norm = obj.total; if (obj.id && obj.store) records.push(obj); } return jsonOut({ok:true, records:records}); } catch(err) { return jsonOut({ok:false, error:err.message, records:[]}); } }
